142    def get_gitstar_ranking(self, owner, repo):
143        """
144        Get the GitStar ranking of the given repository
145        If the repository is not ranked, return None
146        Else, return the ranking
147        """
148        print('Getting GitStar ranking...')
149
150        # Get the needed extra information
151        repo_data = self.get_basic_repository_data(owner, repo)
152
153        # Make sure we got a valid response
154        if repo_data is None:
155            return None
156
157        # Get the language of the repository
158        language = repo_data['language']
159        # Get the stargazer count of the repository
160        stargazer_count = repo_data['stargazers_count']
161
162        # Perform a basic search to list all the repositories in the given language, sorted by their stargazer count
163        ranking_url = f'{constants.BASE_URL_SEARCH}/repositories?q=stars:>0+language:{language}&sort=stars&order=desc&per_page=100'
164        ranking_data = self.try_perform_api_call(ranking_url, constants.SEARCH)
165
166        # Make sure we got a valid response
167        if ranking_data is None:
168            print('Error occured while getting the GitStar ranking.')
169            return None
170
171        # See if the result has multiple pages
172        # If it does, perform a binary search to find the page with the given repository
173        if 'last' in ranking_data.links:
174            # Get the total amount of pages
175            page_count = ranking_data.links['last']['url'].split('=')[-1]
176
177            # Make sure that our repository's stargazer count is above the lowest query-able stargazer count
178            final_page_data = self.try_perform_api_call(
179                ranking_data.links['last']['url'], constants.SEARCH)
180
181            # Make sure we got a valid response
182            if final_page_data is None:
183                print('Error occured while getting the final ranking page.')
184                return None
185
186            # Get the stargazer count of the lsat entry on the last page
187            final_repo_star_count = final_page_data.json()[
188                'items'][-1]['stargazers_count']
189            # If the our count is lower, return none
190            if final_repo_star_count > stargazer_count:
191                return None
192
193            # Perform binary searching to find the correct page
194            lower_bound = 0
195            upper_bound = int(page_count)
196            base_url = ranking_url + '&page='
197            while True:
198                # Calculate the index of the middle page, and get the data from that page
199                middle_page_number = int((upper_bound + lower_bound) / 2)
200                ranking_url = base_url + str(middle_page_number)
201                page_data_response = self.try_perform_api_call(
202                    ranking_url, constants.SEARCH)
203
204                # Make sure we got a valid response
205                if page_data_response is None:
206                    print('Error occured while getting the ranking page.')
207                    return None
208
209                # Get the JSON information out of the response
210                page_data = page_data_response.json()
211
212                # Get the highest and lowest stargazer counts from the query
213                page_highest_star_count = page_data['items'][0]['stargazers_count']
214                page_lowest_star_count = page_data['items'][-1]['stargazers_count']
215
216                # See if the wanted stargazer count is within the bounds of the current page
217                if page_highest_star_count >= stargazer_count and page_lowest_star_count <= stargazer_count:
218                    # If so, find the repository in the page
219                    for index in range(len(page_data['items'])):
220                        if page_data['items'][index]['full_name'] == owner + '/' + repo:
221                            return index + (middle_page_number - 1) * 100
222
223                # If we didn't find the correct page, update the bounds
224                if page_lowest_star_count > stargazer_count:
225                    lower_bound = middle_page_number
226                elif page_highest_star_count < stargazer_count:
227                    upper_bound = middle_page_number
228
229                # See if we have reached the end of the pages
230                if upper_bound - lower_bound <= 1:
231                    return None
232        # Else, the repository should be on the first page
233        else:
234            # Find the repository in the page
235            for current_repo in ranking_data.json()['items']:
236                if current_repo['full_name'] == f'{owner}/{current_repo}':
237                    return current_repo['stargazers_count']
238
239            # If we didn't find the repository, return None
240            return None

369        """
370        Get the total amount of issues that have no responses
371        """
372        print('Getting zero-response issues count...')
373
374        # Get the first page of issues
375        issues_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/issues?per_page=100&state=open&sort=comments&direction=asc'
376        issues_data = self.try_perform_api_call(issues_url, constants.CORE)
377        last_full_no_responses_page = 0
378
379        while True:
380            # Make sure we got a valid response
381            if issues_data is None:
382                print('Error occured while getting the 0-response-issues count.')
383                return None
384
385            # See if there are more pages AND the last issue on this page has no responses
386            # As if that is the case, we need to take a look a the next page
387            if 'next' in issues_data.links and issues_data.json()[-1]['comments'] == 0:
388                last_full_no_responses_page += 1
389                issues_url = issues_data.links['next']['url'] + \
390                    '&per_page=100&state=open&sort=comments&direction=asc'
391                issues_data = self.try_perform_api_call(
392                    issues_url, constants.CORE)
393            # Else if there are no more pages, and the last issue on this page has no responses
394            # Then this whole page must be full of 0-response issues, so return the total number of seen issues
395            # We do last_full_no_responses_page + 1 because it refers to the previous page, and we need to add the current page to it
396            elif issues_data.json()[-1]['comments'] == 0:
397                return (last_full_no_responses_page + 1) * 100
398            # Else, the final 0-responses issue must be somewhere within this page
399            # So we need to find the index of it, and return it summed up with the total number of seen issues
400            else:
401                # Find the index of the final 0-responses issue
402                last_no_response_index = 0
403                for i in range(len(issues_data.json())):
404                    if issues_data.json()[i]['comments'] == 0:
405                        last_no_response_index = i
406                    elif issues_data.json()[i]['comments'] > 0:
407                        break
408
409                # Return the total number of seen issues + the index of the last 0-response issue
410                return (last_full_no_responses_page * 100) + last_no_response_index

429        all_issues = issue_data.json()
430
431        # If there are more pages, add the next 100 issues to the JSON object as well
432        # For now, we only use a maximum of 200 issues, as otherwise we would be making too many API calls
433        if 'next' in issue_data.links:
434            issue_url = issue_data.links['next']['url'] + \
435                '&per_page=100&state=closed&sort=created&direction=desc'

439            if issue_data is None:
440                print('Error occured while getting the average issue resolution time.')
441                return None
442
443            all_issues += issue_data.json()
444
445        # Variable to store the total resolution time in seconds
446        total_resolution_time = 0
447
448        # Loop through all the issues, and add up the time between the issue creation and the issue resolution
449        for issue in all_issues:
450            # Calculate the resolution time
451            creation_date = datetime.strptime(
452                issue['created_at'], '%Y-%m-%dT%H:%M:%SZ')
453            closure_date = datetime.strptime(
454                issue['closed_at'], '%Y-%m-%dT%H:%M:%SZ')
455            issue_resolution_time = closure_date - creation_date
456
457            # Add the resolution time to the total
458            total_resolution_time += issue_resolution_time.seconds
459
460        # Return the average resolution time
461        return total_resolution_time / len(all_issues)

501    def get_release_dates(self, owner, repo, release):
502        """
503        Get the publish date of the given release and the release after it (if it exists)
504        Presumes that the repository
505        """
506
507        # Get the information about the first releases
508        releases_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/releases?per_page=100'
509        releases_data = self.try_perform_api_call(releases_url, constants.CORE)
510        previous_page = None
511
512        while True:
513            # Make sure we got a valid response
514            if releases_data is None:
515                print('Error occured while getting the release dates.')
516                return None
517
518            # Get the json data for easy access
519            releases_json = releases_data.json()
520
521            # See if the current release is within the current page
522            for i in range(len(releases_json)):
523                # If it is, and there is an entry before the current one (aka later in time)
524                # Return the publish dates of both
525                if releases_json[i]['tag_name'] == release and i > 0:
526                    return (releases_json[i]['published_at'], releases_json[i - 1]['published_at'])
527
528                # If it is and there is no entry before the current one, but we have a previous page
529                # Return the publish date of the current previous release on the previous page
530                if releases_json[i]['tag_name'] == release and i == 0:
531                    # If there is no previous page, return None for the previous release
532                    if previous_page is None:
533                        return (releases_json[i]['published_at'], None)
534
535                    # Else return the publish date of the previous release
536                    return (releases_json[i]['published_at'], previous_page.json()[-1]['published_at'])
537
538            # If there is a next page, update previous_page and releases_data
539            if 'next' in releases_data.links:
540                # If the current release is not within the current page, check the next page
541                previous_page = releases_data
542                # Update releases_data by making a new API call to the given next page
543                releases_url = releases_data.links['next']['url'] + \
544                    '&per_page=100'
545                releases_data = self.try_perform_api_call(
546                    releases_url, constants.CORE)
547            # Else, there are no more pages, so return None for both dates
548            else:
549                return (None, None)

588    def try_perform_api_call(self, api_url, call_type):
589        """
590        Perform rate limit checks, and if those pass, perform an API call
591
592        If successful, returns the response
593        If not, returns None
594        """
595
596        # Check if we still have API calls left
597        if self.check_rate_limit(call_type):
598            # Decrement the rate limit of the given type
599            if call_type == constants.CORE:
600                self.core_remaining -= 1
601            elif call_type == constants.SEARCH:
602                self.search_remaining -= 1
603
604            # Return the response, no matter if it succeeded or not
605            return make_api_call(api_url, constants.API_GITHUB)
606
607        # If we don't have any API calls left, return None
608        return None

610    def check_rate_limit(self, call_type):
611        """
612        Function to check if a rate limit has been reached
613
614        Returns true if we can continue with the API call
615        Else it will return false
616        """
617
618        # Check locally if we still have API calls left
619        # If not, check if the rate limits have changed
