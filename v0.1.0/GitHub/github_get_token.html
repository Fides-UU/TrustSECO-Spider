 13class GitHubToken:
 14    """
 15    Class containing the functions needed to obtain a personal GitHub token
 16    """
 17    def authenticate_user(client_id):
 18        """
 19        Try to get a OAuth token for the user.
 20        If unsuccessful, it will return None.
 21        """
 22
 23        # Request a device token from GitHub
 24        # This is used to request an OAuth token for the user.
 25        device_response = GitHubToken.request_device_token_info(client_id)
 26
 27        # See if we succesfully got a device token.
 28        # If not, tell the user, and exit.
 29        if device_response is None:
 30            print("Error: Could not get device token.")
 31            return False
 32
 33        # If we did get a device token, try get the user's OAuth token.
 34        # Print the device verification code so the user can use it to get the OAuth token.
 35        print('User code: ' + device_response['user_code'])
 36        print('Verification URL: ' + device_response['verification_uri'])
 37
 38        # Try to get the user's OAuth token.
 39        user_token = GitHubToken.request_user_token(client_id, device_response)
 40
 41        # See if we got an OAuth token.
 42        # If not, tell the user, and exit.
 43        if user_token is None:
 44            print("Error: Could not get user token.")
 45            return False
 46
 47        # If we did get an OAuth token, return it.
 48        print("Successfully authenticated user.")
 49
 50        # Write the token to the .env file
 51        dotenv.set_key(dotenv_path='.env', key_to_set='GITHUB_TOKEN',
 52                       value_to_set=user_token)
 53
 54        return True

 56    def request_device_token_info(client_id):
 57        """
 58        Request a device token from GitHub.
 59        """
 60        # Request the token using our client ID, with scope of public_repo to only target public repos.
 61        response = requests.post(
 62            'https://github.com/login/device/code',
 63            data={'client_id': client_id},
 64            headers={'Accept': 'application/json'}
 65        )
 66
 67        # See if the post request was successful.
 68        # If not, tell the user, and exit.
 69        if response.status_code != 200:
 70            print("Error: " + response.text)
 71            return None
 72        # If we did get a response, return it
 73        return response.json()
 74
 75    def request_user_token(client_id, device_response):
 76        """Use the device token to get the user's OAuth token."""
 77
 78        # Get some required information from the device response.
 79        # Interval is a github defined integer defining
 80        # the amount of seconds that need to be inbetween the requests.
 81        interval = device_response['interval']
 82        # Our device token
 83        device_token = device_response['device_code']
 84        # Grant type is a github defined string defining the type of token we are requesting.
 85        grant_type = 'urn:ietf:params:oauth:grant-type:device_code'
 86
 87        # Loop until we either get the token, or get an error
 88        while True:
 89            # Get a response from GitHub.
 90            user_token_response_np = requests.post(
 91                'https://github.com/login/oauth/access_token',
 92                data={
 93                    'client_id': client_id,
 94                    'device_code': device_token,
 95                    'grant_type': grant_type
 96                },
 97                headers={
 98                    'Accept': 'application/json'
 99                }
100            )
101
102            # Convert the response to a json object.
103            user_token_response = user_token_response_np.json()
104
105            # See if the response we got was an error.
106            # If so, inform the user of whic error occured, and either sleep or exit.
107            if 'error' in user_token_response:
108                if user_token_response['error'] == 'authorization_pending':
109                    time.sleep(interval)
110                    continue
111
112                if user_token_response['error'] == 'slow_down':
113                    print("Recieved slow-down error.")
114                    interval += 5
115                    time.sleep(interval)
116                    continue
117
118                if user_token_response['error'] == 'expired_token':
119                    print("Error: Token expired.")
120                elif user_token_response['error'] == 'unsupported_grant_type':
121                    print("Error: Unsupported grant type.")
122                elif user_token_response['error'] == 'incorrect_client_credentials':
123                    print("Error: Incorrect client credentials.")
124                elif user_token_response['error'] == 'incorrect_client_credentials':
125                    print("Error: Incorrect client credentials.")
126                elif user_token_response['error'] == 'access_denied':
127                    print("Error: User denied access.")
128                else:
129                    print("Error: " + user_token_response['error'])
130
131                return None
132
133            # If we didn't get an error, return the token.
134            return user_token_response['access_token']
