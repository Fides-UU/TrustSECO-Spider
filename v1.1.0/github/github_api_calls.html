220    def get_gitstar_ranking(self, owner: str, repo: str) -> int:
221        """
222        Get the GitStar ranking of the given repository
223
224        Parameters:
225            owner (str): The owner of the repository
226            repo (str): The repository name
227
228        Returns:
229            int: The GitStar ranking
230        """
231
232        logging.info('Getting GitStar ranking...')
233
234        # Get the needed extra information
235        repository_data = self.get_basic_repository_data(owner, repo)
236
237        # Make sure we got a valid response
238        if repository_data is None:
239            return None
240
241        # Get the language of the repository
242        language = repository_data['language']
243        # Get the stargazer count of the repository
244        stargazer_count = repository_data['stargazers_count']
245
246        # Perform a basic search to list all the repositories in the given language, sorted by their stargazer count
247        ranking_url = f'{constants.BASE_URL_SEARCH}/repositories?q=stars:>0+language:{language}&sort=stars&order=desc&per_page=100'
248        ranking_data = self.try_perform_api_call(ranking_url, constants.SEARCH)
249
250        # Make sure we got a valid response
251        if ranking_data is None:
252            logging.error('Error occurred while getting the GitStar ranking.')
253            return None
254
255        # See if the result has multiple pages
256        # If it does, perform a binary search to find the page with the given repository
257        if 'last' in ranking_data.links:
258            # Get the total amount of pages
259            page_count = ranking_data.links['last']['url'].split('=')[-1]
260
261            # Make sure that our repository's stargazer count is above the lowest query-able stargazer count
262            final_page_data = self.try_perform_api_call(
263                ranking_data.links['last']['url'], constants.SEARCH)
264
265            # Make sure we got a valid response
266            if final_page_data is None:
267                logging.error(
268                    'Error occurred while getting the final ranking page.')
269                return None
270
271            # Get the stargazer count of the lsat entry on the last page
272            final_repo_star_count = final_page_data.json()[
273                'items'][-1]['stargazers_count']
274            # If the our count is lower, return none
275            if final_repo_star_count > stargazer_count:
276                return None
277
278            # Perform binary searching to find the correct page
279            lower_bound = 0
280            upper_bound = int(page_count)
281            base_url = ranking_url + '&page='
282            while True:
283                # Calculate the index of the middle page, and get the data from that page
284                middle_page_number = int((upper_bound + lower_bound) / 2)
285                ranking_url = base_url + str(middle_page_number)
286                page_data_response = self.try_perform_api_call(
287                    ranking_url, constants.SEARCH)
288
289                # Make sure we got a valid response
290                if page_data_response is None:
291                    logging.error(
292                        f'Error occurred while getting ranking page {middle_page_number}.')
293                    return None
294
295                # Get the JSON information out of the response
296                page_data = page_data_response.json()
297
298                # Get the highest and lowest stargazer counts from the query
299                page_highest_star_count = page_data['items'][0]['stargazers_count']
300                page_lowest_star_count = page_data['items'][-1]['stargazers_count']
301
302                # See if the wanted stargazer count is within the bounds of the current page
303                if page_highest_star_count >= stargazer_count and page_lowest_star_count <= stargazer_count:
304                    # If so, find the repository in the page
305                    for index in range(len(page_data['items'])):
306                        if page_data['items'][index]['full_name'] == owner + '/' + repo:
307                            return index + (middle_page_number - 1) * 100
308
309                # If we didn't find the correct page, update the bounds
310                if page_lowest_star_count > stargazer_count:
311                    lower_bound = middle_page_number
312                elif page_highest_star_count < stargazer_count:
313                    upper_bound = middle_page_number
314
315                # See if we have reached the end of the pages
316                if upper_bound - lower_bound <= 1:
317                    return None
318        # Else, the repository should be on the first page
319        else:
320            # Find the repository in the page
321            for current_repo in ranking_data.json()['items']:
322                if current_repo['full_name'] == f'{owner}/{current_repo}':
323                    return current_repo['stargazers_count']
324
325            # If we didn't find the repository, return None
326            return None

364
365        Parameters:
366            owner (str): The owner of the repository
367            repo (str): The repository name
368
369        Returns:
370            int: The total amount of downloads of this repository
371        """

373        logging.info('Getting total download count...')
374

376        releases_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/releases?per_page=100'

400                    releases_url, constants.CORE)
401            # Else stop the loop, as we are done
402            else:
403                break
404
405        # Return the total download count

442        """
443        Get the total amount of issues that have no responses
444
445        Parameters:
446            owner (str): The owner of the repository
447            repo (str): The repository name
448
449        Returns:
450            int: The total amount of issues that have no responses
451        """
452
453        logging.info('Getting zero-response issues count...')
454
455        # Get the first page of issues
456        issues_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/issues?per_page=100&state=open&sort=comments&direction=asc'
457        issues_data = self.try_perform_api_call(issues_url, constants.CORE)
458        last_full_no_responses_page = 0
459
460        while True:
461            # Make sure we got a valid response
462            if issues_data is None:
463                logging.error(
464                    'Error occurred while getting the zero-response-issues count.')
465                return None
466
467            # See if there are more pages AND the last issue on this page has no responses
468            # As if that is the case, we need to take a look a the next page
469            if 'next' in issues_data.links and issues_data.json()[-1]['comments'] == 0:
470                last_full_no_responses_page += 1
471                issues_url = issues_data.links['next']['url'] + \
472                    '&per_page=100&state=open&sort=comments&direction=asc'
473                issues_data = self.try_perform_api_call(
474                    issues_url, constants.CORE)
475            # Else if there are no more pages, and the last issue on this page has no responses
476            # Then this whole page must be full of 0-response issues, so return the total number of seen issues
477            # We do last_full_no_responses_page + 1 because it refers to the previous page, and we need to add the current page to it
478            elif issues_data.json()[-1]['comments'] == 0:
479                return (last_full_no_responses_page + 1) * 100
480            # Else, the final 0-responses issue must be somewhere within this page
481            # So we need to find the index of it, and return it summed up with the total number of seen issues
482            else:
483                # Find the index of the final 0-responses issue
484                last_no_response_index = 0
485                for i in range(len(issues_data.json())):
486                    if issues_data.json()[i]['comments'] == 0:
487                        last_no_response_index = i
488                    elif issues_data.json()[i]['comments'] > 0:
489                        break
490
491                # Return the total number of seen issues + the index of the last 0-response issue
492                return (last_full_no_responses_page * 100) + last_no_response_index

494    def get_average_issue_resolution_time(self, owner: str, repo: str) -> int:
495        """
496        Get the average resolution time of the last 200 issues
497
498        Parameters:
499            owner (str): The owner of the repository
500            repo (str): The repository name
501
502        Returns:
503            int: The average resolution time of the last 200 issues (in seconds)
504        """
505
506        logging.info('Getting average issue resolution time...')
507
508        # Get the first page of closed issues
509        issue_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/issues?per_page=100&state=closed&sort=created&direction=desc'
510        issue_data = self.try_perform_api_call(issue_url, constants.CORE)

512        # Make sure we got a valid response
513        if issue_data is None:
514            logging.error(
515                'Error occurred while getting the average issue resolution time.')
516            return None
517
518        # Add the issues of the first page to the JSON object
519        all_issues = issue_data.json()
520
521        # If there are more pages, add the next 100 issues to the JSON object as well
522        # For now, we only use a maximum of 200 issues, as otherwise we would be making too many API calls
523        if 'next' in issue_data.links:
524            issue_url = issue_data.links['next']['url'] + \
525                '&per_page=100&state=closed&sort=created&direction=desc'
526            issue_data = self.try_perform_api_call(issue_url, constants.CORE)
527
528            # Make sure we got a valid response
529            if issue_data is None:
530                logging.error(
531                    'Error occurred while getting the average issue resolution time.')
532                return None
533
534            all_issues += issue_data.json()
535
536        # Variable to store the total resolution time in seconds
537        total_resolution_time = 0
538
539        # Loop through all the issues, and add up the time between the issue creation and the issue resolution
540        for issue in all_issues:
541            # Calculate the resolution time
542            creation_date = datetime.strptime(
543                issue['created_at'], '%Y-%m-%dT%H:%M:%SZ')
544            closure_date = datetime.strptime(
545                issue['closed_at'], '%Y-%m-%dT%H:%M:%SZ')
546            issue_resolution_time = closure_date - creation_date
547
548            # Add the resolution time to the total
549            total_resolution_time += issue_resolution_time.seconds
550
551        # Return the average resolution time
552        return total_resolution_time / len(all_issues)

599    def get_release_dates(self, owner: str, repo: str, release: str) -> tuple:
600        """
601        Get the publish date of the given release and the release after it (if it exists)
602
603        Parameters:
604            owner (str): The owner of the repository
605            repo (str): The repository name
606            release (str): The release to get the publish dates for
607
608        Returns:
609            tuple: The publish dates of the given release and the release after it (if it exists)
610        """
611
612        logging.info('Getting the publish date of the given release')
613
614        # Get the information about the first releases
615        releases_url = f'{constants.BASE_URL_REPOS}/{owner}/{repo}/releases?per_page=100'
616        releases_data = self.try_perform_api_call(releases_url, constants.CORE)
617        previous_page = None
618
619        while True:
620            # Make sure we got a valid response
621            if releases_data is None:
622                logging.error(
623                    'Error occurred while getting the release dates.')
624                return None
625
626            # Get the json data for easy access
627            releases_json = releases_data.json()
628
629            # See if the current release is within the current page
630            for i in range(len(releases_json)):
631                # If it is, and there is an entry before the current one (aka later in time)
632                # Return the publish dates of both
633                if releases_json[i]['tag_name'] == release and i > 0:
634                    return (releases_json[i]['published_at'], releases_json[i - 1]['published_at'])
635
636                # If it is and there is no entry before the current one, but we have a previous page
637                # Return the publish date of the current previous release on the previous page
638                if releases_json[i]['tag_name'] == release and i == 0:
639                    # If there is no previous page, return None for the previous release
640                    if previous_page is None:
641                        return (releases_json[i]['published_at'], None)
642
643                    # Else return the publish date of the previous release
644                    return (releases_json[i]['published_at'], previous_page.json()[-1]['published_at'])
645
646            # If there is a next page, update previous_page and releases_data
647            if 'next' in releases_data.links:
648                # If the current release is not within the current page, check the next page
649                previous_page = releases_data
650                # Update releases_data by making a new API call to the given next page
651                releases_url = releases_data.links['next']['url'] + \
652                    '&per_page=100'
653                releases_data = self.try_perform_api_call(
654                    releases_url, constants.CORE)
655            # Else, there are no more pages, so return None for both dates
656            else:
657                return (None, None)

734    def try_perform_api_call(self, api_url: str, call_type: str) -> Response:
735        """
736        Perform rate limit checks, and if those pass, perform an API call
737
738        If successful, returns the response
739        If not, returns None
740
741        Parameters:
742            api_url (str): The URL to perform the API call on
743            call_type (str): The type of API call to perform
744
745        Returns:
746            requests.Response: The response from the API call
747        """
748
749        # Check if we still have API calls left
750        if self.check_rate_limit(call_type):
751            # Decrement the rate limit of the given type
752            if call_type == constants.CORE:
753                self.core_remaining -= 1
754            elif call_type == constants.SEARCH:
755                self.search_remaining -= 1
756
757            # Return the response, no matter if it succeeded or not
758            return make_api_call(api_url, constants.API_GITHUB)
759
760        # If we don't have any API calls left, return None
761        return None

763    def check_rate_limit(self, call_type: str) -> bool:
764        """
765        Function to check if a rate limit has been reached
766
767        Parameters:
768            call_type (str): The type of API call to perform
769
770        Returns:
771            bool: Whether or not we can continue with the API call
772        """
773
774        # Check locally if we still have API calls left
775        # If not, check if the rate limits have changed
776        if (call_type == constants.CORE and self.core_remaining == 0) or (call_type == constants.SEARCH and self.search_remaining == 0):
777            # Try to update the rate limit data
778            if self.update_rate_limit_data():
779                # If the update was successful, check the rate limits again
780                if call_type == constants.CORE and self.core_remaining == 0:
781                    logging.warning(
782                        'Core rate-limit reached.')
783                    return False
784                elif call_type == constants.SEARCH and self.search_remaining == 0:
785                    logging.warning(
786                        'Search rate-limit reached.')
787                    return False
788            else:
789                # If the update was unsuccessful, return false
790                return False
791        # If our local count is still above 0, return true
792        return True

221    def get_gitstar_ranking(self, owner: str, repo: str) -> int:
222        """
223        Get the GitStar ranking of the given repository
224
225        Parameters:
226            owner (str): The owner of the repository
227            repo (str): The repository name
228
229        Returns:
230            int: The GitStar ranking
231        """
232
233        logging.info('Getting GitStar ranking...')
234
235        # Get the needed extra information
236        repository_data = self.get_basic_repository_data(owner, repo)
237
238        # Make sure we got a valid response
239        if repository_data is None:
240            return None
241
242        # Get the language of the repository
243        language = repository_data['language']
244        # Get the stargazer count of the repository
245        stargazer_count = repository_data['stargazers_count']
246
247        # Perform a basic search to list all the repositories in the given language, sorted by their stargazer count
248        ranking_url = f'{constants.BASE_URL_SEARCH}/repositories?q=stars:>0+language:{language}&sort=stars&order=desc&per_page=100'
249        ranking_data = self.try_perform_api_call(ranking_url, constants.SEARCH)
250
251        # Make sure we got a valid response
252        if ranking_data is None:
253            logging.error('Error occurred while getting the GitStar ranking.')
254            return None
255
256        # See if the result has multiple pages
257        # If it does, perform a binary search to find the page with the given repository
258        if 'last' in ranking_data.links:
259            # Get the total amount of pages
260            page_count = ranking_data.links['last']['url'].split('=')[-1]
261
262            # Make sure that our repository's stargazer count is above the lowest query-able stargazer count
263            final_page_data = self.try_perform_api_call(
264                ranking_data.links['last']['url'], constants.SEARCH)
265
266            # Make sure we got a valid response
267            if final_page_data is None:
268                logging.error(
269                    'Error occurred while getting the final ranking page.')
270                return None
271
272            # Get the stargazer count of the lsat entry on the last page
273            final_repo_star_count = final_page_data.json()[
274                'items'][-1]['stargazers_count']
275            # If the our count is lower, return none
276            if final_repo_star_count > stargazer_count:
277                return None
278
279            # Perform binary searching to find the correct page
280            lower_bound = 0
281            upper_bound = int(page_count)
282            base_url = ranking_url + '&page='
283            while True:
284                # Calculate the index of the middle page, and get the data from that page
285                middle_page_number = int((upper_bound + lower_bound) / 2)
286                ranking_url = base_url + str(middle_page_number)
287                page_data_response = self.try_perform_api_call(
288                    ranking_url, constants.SEARCH)
289
290                # Make sure we got a valid response
291                if page_data_response is None:
292                    logging.error(
293                        f'Error occurred while getting ranking page {middle_page_number}.')
294                    return None
295
296                # Get the JSON information out of the response
297                page_data = page_data_response.json()
298
299                # Get the highest and lowest stargazer counts from the query
300                page_highest_star_count = page_data['items'][0]['stargazers_count']
301                page_lowest_star_count = page_data['items'][-1]['stargazers_count']
302
303                # See if the wanted stargazer count is within the bounds of the current page
304                if page_highest_star_count >= stargazer_count and page_lowest_star_count <= stargazer_count:
305                    # If so, find the repository in the page
306                    for index in range(len(page_data['items'])):
307                        if page_data['items'][index]['full_name'] == owner + '/' + repo:
308                            return index + (middle_page_number - 1) * 100
309
310                # If we didn't find the correct page, update the bounds
311                if page_lowest_star_count > stargazer_count:
312                    lower_bound = middle_page_number
313                elif page_highest_star_count < stargazer_count:
314                    upper_bound = middle_page_number
315
316                # See if we have reached the end of the pages
317                if upper_bound - lower_bound <= 1:
318                    return None
319        # Else, the repository should be on the first page
320        else:
321            # Find the repository in the page
322            for current_repo in ranking_data.json()['items']:
323                if current_repo['full_name'] == f'{owner}/{current_repo}':
324                    return current_repo['stargazers_count']
325
326            # If we didn't find the repository, return None
327            return None
