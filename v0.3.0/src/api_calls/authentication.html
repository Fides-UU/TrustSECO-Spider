 15    # Make sure that the .env file exists, and has the proper key-values

 26def authenticate_user():
 27    """
 28    Authenticate the user for the GitHub and Libraries.io APIs
 29    """
 30    # (Re)load the environment variables before authenticating
 31    # To make sure that we can write tokens to the .env file

 38def gh_authenticate_user():
 39    """
 40    Authenticate the user for the GitHub API
 41    """
 42
 43    # Return true if the token is already present
 44    if os.getenv(constants.GITHUB_TOKEN) is None or os.getenv(constants.GITHUB_TOKEN) == '':
 45        print('Starting GitHub authentication process')
 46
 47        while True:
 48            # See if the authentication process succeeded
 49            # If so, continue with the API call
 50            if gh_get_token(constants.GITHUB_ORG_CODE):
 51                print('Successfully authenticated user')
 52                # Reload the environment variables
 53                # As otherwise the GitHub token would not have been updated
 54                load_dotenv(dotenv_path=constants.ENVIRON_FILE, override=True)
 55
 56                break
 57            # If not, stop the program
 58            else:
 59                print('Authentication failed.')

 62def gh_get_token(client_id):
 63    """
 64    Try to get a OAuth token for the user.
 65    If unsuccessful, it will return None.
 66    """
 67
 68    # Request a device token from GitHub
 69    # This is used to request an OAuth token for the user.
 70    try:
 71        # Request the token using our client ID, with scope of public_repo to only target public repos.
 72        device_response = requests.post(
 73            'https://github.com/login/device/code',
 74            data={'client_id': client_id},
 75            headers={'Accept': 'application/json'}
 76        )
 77    except requests.exceptions.RequestException as error:
 78        print('Requests encountered an error:')
 79        print(error)
 80        return False
 81
 82    # Make sure the request was successful
 83    if device_response.status_code == 200:
 84        print('Successfully received device token')
 85        device_data = device_response.json()
 86    else:
 87        print('Failed to receive device token')
 88        return False
 89
 90    # If we did get a device token, try get the user's OAuth token.
 91    # Print the device verification code so the user can use it to get the OAuth token.
 92    print('User code: ' + device_data['user_code'])
 93    print('Verification URL: ' + device_data['verification_uri'])
 94
 95    # Try to get the user's OAuth token.
 96    return request_user_token(
 97        client_id, device_data['interval'], device_data['device_code'])

100def request_user_token(client_id, interval, device_token):
101    """Use the device token to get the user's OAuth token."""
102
103    # Grant type is a github defined string defining the type of token we are requesting.
104    grant_type = 'urn:ietf:params:oauth:grant-type:device_code'
105
106    # Loop until we either get the token, or get an error
107    while True:
108        # Get a response from GitHub.
109        user_token_response_np = requests.post(
110            'https://github.com/login/oauth/access_token',
111            data={
112                'client_id': client_id,
113                'device_code': device_token,
114                'grant_type': grant_type
115            },
116            headers={
117                'Accept': 'application/json'
118            }
119        )
120
121        # Convert the response to a json object.
122        user_token_response = user_token_response_np.json()
123
124        # See if the response we got was an error.
125        # If so, inform the user of whic error occured, and either sleep or exit.
126        if 'error' in user_token_response:
127            if user_token_response['error'] == 'authorization_pending':
128                time.sleep(interval)
129                continue
130
131            if user_token_response['error'] == 'slow_down':
132                print("Recieved slow-down error.")
133                interval += 5
134                time.sleep(interval)
135                continue
136
137            if user_token_response['error'] == 'expired_token':
138                print("Error: Token expired.")
139            elif user_token_response['error'] == 'unsupported_grant_type':
140                print("Error: Unsupported grant type.")
141            elif user_token_response['error'] == 'incorrect_client_credentials':
142                print("Error: Incorrect client credentials.")
143            elif user_token_response['error'] == 'incorrect_client_credentials':
144                print("Error: Incorrect client credentials.")
145            elif user_token_response['error'] == 'access_denied':
146                print("Error: User denied access.")
147            else:
148                print("Error: " + user_token_response['error'])
149
150            return False
151
152        # If we did get an OAuth token, return it.
153        print("Successfully authenticated user.")
154
155        # Write the token to the .env file
156        set_key(dotenv_path=constants.ENVIRON_FILE, key_to_set='GITHUB_TOKEN',
157                value_to_set=user_token_response['access_token'])
158
159        return True

162def lib_authenticate_user():
163    """

167    # Return true if the token is already present

172            print('Please enter your token:')
173
174            token = input()
175
176            # See if the user entered a valid token
177            # by making sure text has been entered
178            # and that the entered text is alphanumeric
179            if len(token) > 0 and token.isalnum():
180                print('Received API token')
181                """We should probably test this token via some kind of quick api call"""
182
183                # Write the token to the .env file

185
186                # Reload the environment variables
187                # As otherwise the environmental tokens would not have been updated
188                load_dotenv(dotenv_path=constants.ENVIRON_FILE, override=True)
189
190                break
191            # If not, stop the program
192            else:
193                print('Authentication failed.')
