 15    def get_repository_user_count(self, owner, repo):
 16        """
 17        Gets the number of users of a given repository
 18        """
 19
 20        # Create the URL for the repository
 21        main_page_url = f'https://github.com/{owner}/{repo}'
 22
 23        # Get the main page
 24        main_page = requests.get(main_page_url)
 25
 26        # Make sure the main page is valid
 27        if main_page.status_code != 200:
 28            return None
 29
 30        # Create a BeautifulSoup object
 31        soup = BeautifulSoup(main_page.text, 'html.parser')
 32
 33        # Get the <a> tags that might contain our number
 34        possible_links = soup.find_all(
 35            'a', class_='Link--primary no-underline')
 36
 37        # Initialise the counter to None
 38        user_count = None
 39        # Go through all the <a> tags and find the one containing "Used by"
 40        # as this is the one containing the number of users
 41        for link in possible_links:
 42            if 'used by' in link.text.lower():
 43                value = link.span.get('title')
 44                if value is not None:
 45                    # Remove unwanted characters
 46                    value = value.replace(',', '').replace('\n', '').strip()
 47                    # Convert to integer
 48                    user_count = int(value.split(' ')[0])
 49                    # Break as we have found our wanted node
 50                    break
 51
 52        return user_count
